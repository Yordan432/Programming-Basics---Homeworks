using System;

namespace Sunglasses
{
    class Sunglasses
    {
        static void Main()
        {
            int n = int.Parse(Console.ReadLine());
            var colons = 5 * n;

            for (int row = 0; row <n; row++)
            {
                for (int col = 0; col <colons; col++)
                {
                    Console.Write("*");
                    Console.WriteLine();
                }
            }
            
        }
    }
}
