using System;
 
namespace currencyConvector
{
    class currencyConvector
    {
        static void Main()
        {
            var amo = double.Parse(Console.ReadLine());
            var input = Console.ReadLine(); // BGN EUR USD GBP
            var output = Console.ReadLine(); // GBN EUR USD GBP
            var lev = 1.00000;
            var eur = 1.95583;
            var usd = 1.79549;
            var gbp = 2.53405;            
            switch (input)
            {
                case "BGN":
                    amo = amo * lev;
                    break;
                case "EUR":
                    amo = amo * eur;
                    break;
                case "USD":
                    amo = amo * usd;
                    break;
                case "GBP":
                    amo = amo * gbp;
                    break;
                default:
                    break;
            }
            switch (output)
            {
                case "BGN":
                    amo = amo / lev;
                    break;
                case "EUR":
                    amo = amo / eur;
                    break;
                case "USD":
                    amo = amo / usd;
                    break;
                case "GBP":
                    amo = amo / gbp;
                    break;
                default:
                    break;
            }
            Console.WriteLine(Math.Round(amo, 2));
 
        }
    }
}
