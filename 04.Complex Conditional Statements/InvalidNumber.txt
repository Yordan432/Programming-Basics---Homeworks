using System;

namespace InvalidNumber
{
    class InvalidNumber
    {
        static void Main(string[] args)
        {
            int number = int.Parse(Console.ReadLine());

            if(number <=100)
            {
                if(number <200 && number >=220)
                {
                    if(number < 0)
                    {
                        Console.WriteLine("Invalid");
                    }
                    Console.WriteLine("Invalid");
                }
            }
        }
    }
}
