using System;

namespace BonusScore
{
    class BonusScore
    {
        static void Main(string[] args)
        {
            Console.Write("Въведете точки : ");
            double number = double.Parse(Console.ReadLine());
            double BonusScore = 0;

            Console.Write("бонус точките са : " );
            if(number <=100)
            {
                BonusScore = 5;
            }
            else if (number > 1000)
            {
                BonusScore = 0.1 * number;
            }
            else if(number > 100)
            {
                BonusScore = 0.2 * number;
            }
            if(number % 2 ==0)
            {
                BonusScore += 1;
            }
            else if(number % 10 ==5)
            {
                BonusScore += 2;
            }

            Console.WriteLine(BonusScore);
            Console.Write("Oбщ брой точки : ");
            Console.WriteLine(BonusScore + number);

        }
    }
}
